let angleSlider, pressureSlider, volumeSlider; let jarAngle = 0; let stabilityText = ""; function setup() { createCanvas(800, 500); // 角度控制（傾斜） angleSlider = createSlider(0, 90, 0); angleSlider.position(20, 20); // 壓力控制（外力） pressureSlider = createSlider(0, 100, 50); pressureSlider.position(20, 50); // 容量控制（重心影響） volumeSlider = createSlider(0, 100, 50); volumeSlider.position(20, 80); } function draw() { background(240); let angle = angleSlider.value(); let pressure = pressureSlider.value(); let volume = volumeSlider.value(); // ===== 力學邏輯（簡化模型）===== let instability = angle + pressure * 0.3 - volume * 0.2; if (instability > 60) { jarAngle = lerp(jarAngle, radians(45), 0.05); stabilityText = "⚠️ 不穩定：即將傾倒"; } else if (instability > 30) { jarAngle = lerp(jarAngle, radians(15), 0.05); stabilityText = "⚠️ 輕微晃動"; } else { jarAngle = lerp(jarAngle, 0, 0.05); stabilityText = "✅ 穩定狀態"; } // ===== 畫調味罐 ===== push(); translate(width / 2, height / 2); rotate(jarAngle); fill(200, 180, 150); rect(-40, -80, 80, 160, 10); fill(150, 100, 60); rect(-30, -100, 60, 30, 5); pop(); // ===== UI文字 ===== fill(0); textSize(14); text("傾斜角度: " + angle, 200, 30); text("外部壓力: " + pressure, 200, 60); text("內容容量: " + volume, 200, 90); textSize(18); text(stabilityText, 20, 150); // 成功 / 失敗提示 if (instability > 60) { textSize(22); fill(200, 0, 0); text("❌ 失敗：調味罐翻倒", 20, 200); } else if (instability < 20) { fill(0, 150, 0); text("🏆 成功：穩定操作完成", 20, 200); } }
